adjustments are needed.

 export LD_LIBRARY_PATH=${LIBRARYPATH4FEDORA}

 export LD_LIBRARY_PATH=${LIBRARYPATH4OPENSUSE}
 export CLASSPATH=${CLASSPATH4OPENSUSE}
